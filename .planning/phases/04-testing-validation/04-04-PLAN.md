    callback = AsyncMock()
    
    # Prepare raw lines to be "received" from socket
    lines = [
        "15.03.26 10:15:00;RING;0;0123456789;987654321;SIP0",
        "15.03.26 10:15:00;RING;0;0123456789;987654321;SIP0",  # duplicate
    ]
    
    # Mock socket to yield lines then exit
    async def mock_socket_stream():
        for line in lines:
            yield line.encode() + b'\n'
    
    # Patch the socket.socket or use a context manager
    with patch('socket.socket') as mock_socket_class:
        mock_socket = MagicMock()
        mock_socket_class.return_value = mock_socket
        # ... configure mock to return lines
        
        # Run adapter for short time with timeout
        try:
            await asyncio.wait_for(adapter.start(callback), timeout=1.0)
        except asyncio.TimeoutError:
            pass  # Expected — adapter loops forever
    
    # Verify callback was called with 2 events
